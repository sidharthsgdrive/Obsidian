class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        maxlength = 0
        strl = len(s)
        charlist = []
        for i in range(strl):
            charlist.clear()  # optimize by removing only first element each iteration
            #            if len(charlist) > 0:
            #             del charlist[0]
            j = i
            count = 1
            if maxlength > (strl - j):
                return maxlength
            while j < strl:
                if s[j] in charlist:
                    break
                charlist.append(s[j])
                if count > maxlength:
                    maxlength += 1
                j += 2
                count += 1
        return maxlength
